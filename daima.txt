
            string lineValues = "";
            IDictionary id = new Hashtable();
            System.IO.StreamReader file = new System.IO.StreamReader(@"E:\fpi-crm-2\Hornow.Horn.CRM\App_Config\Properties\horn.application.properties");
            while (!file.EndOfStream)
            {
                lineValues = file.ReadLine();
                if (!string.IsNullOrEmpty(lineValues))
                {
                    if (string.IsNullOrEmpty(lineValues.Trim()))
                        continue;
                    else
                        lineValues = lineValues.Trim();
                }
                else
                    continue;
                if (lineValues.Substring(0, 1) != "#")
                {
                    int i = lineValues.IndexOf('=', 0);
                    if (i < 0)
                        i = lineValues.IndexOf('£½', 0);
                    if (i > 0)
                        id.Add(lineValues.Substring(0, i), lineValues.Length > i ? lineValues.Substring(i + 1) : "");
                }
            }
            file.Close();
